Customized Plugin: Styled

Code

$('#TwitterWidget3').TwitterFeed({ 
    userName: 'degdigital', 
    limit: 5,
    feedLocation: 'data/twitter.json',
    userHeader: function(o, data){
        return '<div class="' + o.classPrefix + '-header">\nCustomize the output using javascript templates</div>';
    },
    listItemTemplate: function(o, data) {
        return '<li class="' + o.classPrefix + '-item ' + o.classPrefix + '-item-' + data.index + '">\n'
            +  '<span class="' + o.classPrefix + '-text">' + data.text + '</span> '
            +  '</li>\n';
    },
    userFooter: function(o, data){
        return '<div class="' + o.classPrefix + '-footer">\n'
         + 'Customize the output using javascript templates</div>';
    }
});
